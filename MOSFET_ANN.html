# feedforward function
def feed_forward(A0):
  # layer 1 calculations
  Z1 = W1 @ A0 + b1
  A1 = sigmoid(Z1)

  # layer 2 calculations
  Z2 = W2 @ A1 + b2
  A2 = sigmoid(Z2)

  # layer 3 calculations
  Z3 = W3 @ A2 + b3
  A3 = sigmoid(Z3)

  cache = {
      #we need those for backpropagation calculations
      "A0": A0,
      "A1": A1,
      "A2": A2
  }
  return A3, cache

import numpy as np

def cost(y_pred, y_true, threshold_low=1e-5,
         high_weight=10.0, low_weight=1.0):
    """
    Compute a weighted Mean Squared Error where samples with
    y_true < threshold_low OR y_true > threshold_high are weighted more.

    Parameters:
        y_true (array-like): Ground truth values
        y_pred (array-like): Predicted values
        threshold_low (float): Lower value below which higher weight is applied
        high_weight (float): Weight for extreme y_true values
        low_weight (float): Weight for mid-range y_true values

    Returns:
        float: Weighted MSE
    """
    y_true = np.array(y_true)
    y_pred = np.array(y_pred)

    errors = y_pred - y_true
    squared_errors = errors ** 2

    # Weight more if target is outside [threshold_low, threshold_high]
    weights = np.where(y_true < threshold_low,
                       high_weight, low_weight)

    weighted_mse =  np.mean(weights * squared_errors)
    return weighted_mse

import numpy as np

def backprop_layer_3(y_hat, Y, A2, W3, m, threshold_low=1e-5,
                     high_weight=10.0, low_weight=1.0):
    A3 = y_hat  # final output

    # Step 0: Create sample-wise weights (element-wise logical OR)
    weights = np.where(Y < threshold_low, high_weight, low_weight)

    # Step 1: Compute dC/dZ3 using weighted MSE (with sigmoid output)
    dC_dZ3 = (2 / m) * weights * (A3 - Y) * A3 * (1 - A3)

    # Step 2: dC/dW3 = dC_dZ3 @ A2.T
    dC_dW3 = dC_dZ3 @ A2.T

    # Step 3: dC/db3 = sum across samples
    dC_db3 = np.sum(dC_dZ3, axis=1, keepdims=True)

    # Step 4: dC/dA2 = W3^T @ dC_dZ3
    dC_dA2 = W3.T @ dC_dZ3

    return dC_dW3, dC_db3, dC_dA2

def backprop_layer_2(propagator_dC_dA2, A1, A2, W2, m):

  # step 1. calculate dC/dZ2 = dC/dA2 * dA2/dZ2

  # use sigmoid derivation to arrive at this answer:
  #   sigmoid'(z) = sigmoid(z) * (1 - sigmoid(z))
  #     and if a = sigmoid(z), then sigmoid'(z) = a * (1 - a)
  dA2_dZ2 = A2 * (1 - A2)
  dC_dZ2 = propagator_dC_dA2 * dA2_dZ2

  # step 2. calculate dC/dW2 = dC/dZ2 * dZ2/dW2
  dZ2_dW2 = A1
  dC_dW2 = dC_dZ2 @ dZ2_dW2.T

  # step 3. calculate dC/db2 = np.sum(dC/dZ2, axis=1, keepdims=True)
  dC_db2 = np.sum(dC_dW2, axis=1, keepdims=True)

  # step 4. calculate propagator dC/dA1 = dC/dZ2 * dZ2/dA1
  dZ2_dA1 = W2
  dC_dA1 = W2.T @ dC_dZ2

  return dC_dW2, dC_db2, dC_dA1

def backprop_layer_1(propagator_dC_dA1, A1, A0, W1, m):

  # step 1. calculate dC/dZ1 = dC/dA1 * dA1/dZ1

  # use sigmoid derivation to arrive at this answer:
  #   sigmoid'(z) = sigmoid(z) * (1 - sigmoid(z))
  #     and if a = sigmoid(z), then sigmoid'(z) = a * (1 - a)
  dA1_dZ1 = A1 * (1 - A1)
  dC_dZ1 = propagator_dC_dA1 * dA1_dZ1
  assert dC_dZ1.shape == (n[1], m)


  # step 2. calculate dC/dW1 = dC/dZ1 * dZ1/dW1
  dZ1_dW1 = A0
  assert dZ1_dW1.shape == (n[0], m)

  dC_dW1 = dC_dZ1 @ dZ1_dW1.T
  assert dC_dW1.shape == (n[1], n[0])

  # step 3. calculate dC/db1 = np.sum(dC/dZ1, axis=1, keepdims=True)
  dC_db1 = np.sum(dC_dW1, axis=1, keepdims=True)
  assert dC_db1.shape == (n[1], 1)

  return dC_dW1, dC_db1

def train():
    global W3, W2, W1, b3, b2, b1

    epochs = 16000
    alpha = 0.1
    costs = []

    for e in range(epochs):
        # 1. FEED FORWARD
        y_hat, cache = feed_forward(A0)

        # 2. COST
        error = cost(y_hat, Y)
        costs.append(error)

        m = Y.shape[1]  # define batch size

        # 3. BACKPROP
        dC_dW3, dC_db3, dC_dA2 = backprop_layer_3(
            y_hat, Y,
            A2=cache["A2"],
            W3=W3,m=m
        )

        dC_dW2, dC_db2, dC_dA1 = backprop_layer_2(
            propagator_dC_dA2=dC_dA2,
            A1=cache["A1"],
            A2=cache["A2"],
            W2=W2,m=m
        )

        dC_dW1, dC_db1 = backprop_layer_1(
            propagator_dC_dA1=dC_dA1,
            A1=cache["A1"],
            A0=cache["A0"],
            W1=W1,m=m
        )

        # 4. UPDATE WEIGHTS AND BIASES
        W3 -= alpha * dC_dW3
        W2 -= alpha * dC_dW2
        W1 -= alpha * dC_dW1

        b3 -= alpha * dC_db3
        b2 -= alpha * dC_db2
        b1 -= alpha * dC_db1

        # 5. PRINT COST PERIODICALLY
        if e % 100 == 0:
            print(f"Epoch {e}: cost = {error:.6f}")

    return costs

def trainBatch():
    global W3, W2, W1, b3, b2, b1

    epochs = 40
    alpha = 0.1
    batch_size = 8
    costs = []
    costs.append(cost(feed_forward(A0)[0], Y))
    for e in range(epochs):
        # ---- Shuffle for mini-batch training ----
        permutation = np.random.permutation(Y.shape[1])
        A0_shuffled = A0[:, permutation]
        Y_shuffled = Y[:, permutation]

        for i in range(0, Y.shape[1], batch_size):
            A0_batch = A0_shuffled[:, i:i+batch_size]
            Y_batch = Y_shuffled[:, i:i+batch_size]
            m = Y_batch.shape[1]

            # ---- Forward pass ----
            y_hat, cache = feed_forward(A0_batch)

            # ---- Backpropagation ----
            dC_dW3, dC_db3, dC_dA2 = backprop_layer_3(
                y_hat, Y_batch, A2=cache["A2"], W3=W3, m=m
            )

            dC_dW2, dC_db2, dC_dA1 = backprop_layer_2(
                propagator_dC_dA2=dC_dA2,
                A1=cache["A1"], A2=cache["A2"], W2=W2, m=m
            )

            dC_dW1, dC_db1 = backprop_layer_1(
                propagator_dC_dA1=dC_dA1,
                A1=cache["A1"], A0=cache["A0"], W1=W1, m=m
            )

            # ---- Parameter update ----
            W3 -= alpha * dC_dW3
            W2 -= alpha * dC_dW2
            W1 -= alpha * dC_dW1

            b3 -= alpha * dC_db3
            b2 -= alpha * dC_db2
            b1 -= alpha * dC_db1

        # ----  Full forward pass on entire dataset ----
        full_y_hat, _ = feed_forward(A0)
        full_cost = cost(full_y_hat, Y)  # Mean over all samples

        costs.append(full_cost)

        if e % 5 == 0:
            print(f"Epoch {e}: total dataset cost = {full_cost:.6f}")

    return costs
